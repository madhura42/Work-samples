mean_df = df.groupby('location_id').agg('mean').reset_index()
mean_df['mean_rating_scaled'] = scaler.fit_transform(mean_df[['rating']].to_numpy())
mean_df = mean_df[['location_id', 'mean_rating_scaled']]

count_df = df.groupby('location_id').agg('count').reset_index()
count_df['count_rating_scaled'] = scaler.fit_transform(count_df[['rating']].to_numpy())
count_df = count_df[['location_id', 'count_rating_scaled']]

rank_df = mean_df.merge(count_df, on='location_id')

# Take euclidean norm
rank_df['euclidean']= np.sqrt(rank_df['mean_rating_scaled']**2+rank_df['count_rating_scaled']**2)

# Rank the values
rank_df['rank'] = rank_df['euclidean'].rank(ascending=False)
rank_df['rank'] = rank_df['rank'].astype('int')
rank_df = rank_df.sort_values(by=['rank'])

# Traning the model with CV
def train_cv( X_train, y_train, model_name, model, cv=5):
    kf = KFold(n_splits=cv,random_state=42, shuffle=True)
    accuracy_list = []
    precision_list = []
    recall_list =[]
    f1score_list =[]
    i=0
    print(model_name)
    print("Performing training with cross validation for ", model_name)
    for train_index, test_index in kf.split(X_train):
        X_cv_train, X_cv_test = X_train.iloc[train_index], X_train.iloc[test_index]
        y_cv_train, y_cv_test = y_train.iloc[train_index], y_train.iloc[test_index]
        y_cv_train = y_cv_train.to_numpy().ravel()
        y_cv_test = y_cv_test.to_numpy().ravel()
        model = model.fit(X_cv_train,y_cv_train)
        y_cv_pred = model.predict(X_cv_test)
        accuracy = accuracy_score(y_cv_test, y_cv_pred)
        precision = precision_score(y_cv_test, y_cv_pred, average='weighted')
        recall = recall_score(y_cv_test, y_cv_pred, average='weighted')
        f1score = f1_score(y_cv_test, y_cv_pred, average='weighted')

        print("Cross validation:{} | Accuracy: {:.4f} | Precision: {:.4f} | Recall: {:.4f} | F-1 score: {:.4f}".format(i+1, accuracy, precision, recall, f1score))
        accuracy_list.append(accuracy)
        precision_list.append(precision)
        recall_list.append(recall)
        f1score_list.append(f1score)
        i+=1
    print('Mean CV Accuracy :', (sum(accuracy_list)/cv))
    print('Mean CV Precision :', (sum(precision_list)/cv))
    print('Mean CV Recall :', (sum(recall_list)/cv))
    print('Mean CV F1-score :', (sum(f1score_list)/cv))
    return model

# Predicting on test set
def predict(model_name, model, X_test, y_test):
    y_test = y_test.to_numpy().ravel()
    print("Predicting on test set for ", model_name)
    y_pred = model.predict(X_test)
    print('Accuracy test:',accuracy_score(y_test, y_pred))
    print('Precision test:' ,precision_score(y_test, y_pred, average='weighted'))
    print('Recall test:' ,recall_score(y_test, y_pred, average='weighted'))
    print('F1-score test:' ,f1_score(y_test, y_pred, average='weighted'))
    return y_test, y_pred

#Utility function to evaluate model predictions against testing data
def model_evaluation(y_pred, y_test):
    print("Evaluating model performance...", "\n")

    cnf_mat = confusion_matrix(y_test,y_pred)
    sns.heatmap(cnf_mat.T, square=True, annot=True, fmt='d', cmap='Blues', cbar=False)
    plt.title('Confusion Matrix')
    plt.xlabel('True label')
    plt.ylabel('Predicted label')
    plt.show()
    
    print('Accuracy: {}%'.format(round(accuracy_score(y_test, y_pred)*100, 2)))
    print('Precision: {}%'.format(round(precision_score(y_test, y_pred, average='weighted')*100, 2)))
    print('Recall: {}%'.format(round(recall_score(y_test, y_pred, average='weighted')*100, 2)))
    print('F1 Score: {}'.format(round(f1_score(y_test, y_pred, average='weighted'), 1)))
